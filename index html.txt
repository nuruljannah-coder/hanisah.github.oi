    <script>
        // Init Lucide Icons
        lucide.createIcons();

        // Constants of the model
        const H_max = 15.0;
        const k = 0.22;
        const x_0 = 5.5;

        // Calculate value of f(x) - Height
        function f(x) {
            return H_max / (1 + Math.exp(-k * (x - x_0)));
        }

        // Calculate value of f'(x) - Derivative (Growth velocity)
        function f_prime(x) {
            const expTerm = Math.exp(-k * (x - x_0));
            const numerator = H_max * k * expTerm;
            const denominator = Math.pow(1 + expTerm, 2);
            return numerator / denominator;
        }

        // Generate data arrays for plotting from Year 1 to 25
        const ageLabels = [];
        const heightData = [];
        const velocityData = [];

        for (let x = 1; x <= 25; x += 0.5) {
            ageLabels.push(x.toFixed(1));
            heightData.push(f(x));
            velocityData.push(f_prime(x));
        }

        // Setup dynamic tangent line data points based on current selected X
        function getTangentDatasetPoints(selectedX) {
            const yVal = f(selectedX);
            const slope = f_prime(selectedX);
            
            // Generate a line segment centered around selectedX with length ~4 years
            const startX = Math.max(1, selectedX - 2.5);
            const endX = Math.min(25, selectedX + 2.5);
            
            const points = [];
            for (let x = startX; x <= endX; x += 0.5) {
                // Point-slope equation of tangent: y - y1 = m(x - x1)  =>  y = m(x - x1) + y1
                const tangentY = slope * (x - selectedX) + yVal;
                points.push({ x: x, y: tangentY });
            }
            return points;
        }

            } else {
                return `🚨 <strong>Senescent Phase & Replanting (Years 18-25):</strong> The palm has reached ${h.toFixed(1)}m, approaching the asymptotic ceiling of 15m. Growth rate is almost static ($f'(x) \\approx ${hPrime.toFixed(2)}$ m/year). Harvesting is inefficient, expensive, and dangerous. Implement mechanical clearing and prepare for the next replanting cycle.`;
            }
        }

        // Dynamically recalculate equations, update UI components, and adjust SVG graphics
        function updateSimulator(val) {
            const x = parseFloat(val);
            const h = f(x);
            const hPrime = f_prime(x);

            // Update UI Labels
            lblAge.innerText = x.toFixed(1);
            valHeight.innerText = `${h.toFixed(2)}m`;
            valDerivative.innerText = `${hPrime.toFixed(3)} m/yr`;
            lblLiveHeight.innerText = `Height: ${h.toFixed(2)}m`;

            // LaTeX Text updates inside math modules (re-calculating math variables dynamically)
            const latexHeightElem = document.getElementById('latexHeight');
            const latexDerivativeElem = document.getElementById('latexDerivative');
            
            if (typeof katex !== 'undefined') {
                katex.render(`f(${x.toFixed(1)}) = \\frac{15}{1 + e^{-0.22(${x.toFixed(1)} - 5.5)}} = ${h.toFixed(2)}\\text{ m}`, latexHeightElem, { throwOnError: false });
                katex.render(`f'(${x.toFixed(1)}) = \\frac{3.3 e^{-0.22(${x.toFixed(1)} - 5.5)}}{(1+e^{-0.22(${x.toFixed(1)} - 5.5)})^2} = ${hPrime.toFixed(3)}\\text{ m/yr}`, latexDerivativeElem, { throwOnError: false });
            }

            // Adjust Layman Agrotech Advice
            lblLaymanAdvice.innerHTML = getAgronomicAdvice(x, h, hPrime);

            // Re-render the Dynamic Tangent Line on the chart
            if (myChart) {
                // Generate absolute line positions matching the age scale indices
                const tangentPoints = getTangentDatasetPoints(x);
                
                // Align the points back to our chart indexes
                const alignedTangentData = ageLabels.map(labelAgeStr => {
                    const labelAge = parseFloat(labelAgeStr);
                    // Find a calculated tangent value within matching range
                    const matchedPoint = tangentPoints.find(p => Math.abs(p.x - labelAge) < 0.05);
                    return matchedPoint ? matchedPoint.y : null;
                });
                
                myChart.data.datasets[2].data = alignedTangentData;
                myChart.update('none'); // silent updates
            }

            // Animate SVG Oil Palm Graphic height scaling dynamically
            // Standard trunk height is 115px at 15 meters, so let's scale the Y scale.
            // Map height scale: 0m to 15m corresponding to SVG trunk height
            const maxValSVG = 15;
            const ratio = h / maxValSVG;
            
            // Apply scale to canopy translation and trunk path length
            const trunkYPos = 115 - (95 * ratio); // 115 is floor, 20 is max ceiling
            
            svgCanopy.setAttribute('transform', `translate(50, ${trunkYPos})`);
            svgTrunk.setAttribute('d', `M 46,115 L 48,${85 + (30*(1-ratio))} Q 50,${45 + (15*(1-ratio))} 48,${trunkYPos} L 52,${trunkYPos} Q 50,${45 + (15*(1-ratio))} 52,${85 + (30*(1-ratio))} L 54,115 Z`);
        }

        // Slide Animation logic
        let animationInterval;
        let isAnimating = false;

        function startAnimation() {
            if (isAnimating) {
                stopAnimation();
                return;
            }
            
            isAnimating = true;
            btnPlay.innerHTML = `<i data-lucide="pause" class="w-3.5 h-3.5"></i><span>Pause Growth</span>`;
            lucide.createIcons();

            let currentAge = parseFloat(sliderAge.value);
            if (currentAge >= 25) {
                currentAge = 1.0;
            }

            animationInterval = setInterval(() => {
                currentAge += 0.2;
                if (currentAge > 25) {
                    currentAge = 25;
                    stopAnimation();
                }
                sliderAge.value = currentAge;
                updateSimulator(currentAge);
            }, 60);
        }

        function stopAnimation() {
            isAnimating = false;
            clearInterval(animationInterval);
            btnPlay.innerHTML = `<i data-lucide="play" class="w-3.5 h-3.5"></i><span>Animate Growth</span>`;
            lucide.createIcons();
        }

        // Action Listeners
        sliderAge.addEventListener('input', (e) => {
            stopAnimation();
            updateSimulator(e.target.value);
        });

        btnPlay.addEventListener('click', startAnimation);
        
        btnReset.addEventListener('click', () => {
            stopAnimation();
            sliderAge.value = 5.5;
            updateSimulator(5.5);
        });

        // Initialize setup on window load
        window.onload = function() {
            // Safely execute auto-render after DOM is fully constructed
            if (typeof renderMathInElement !== 'undefined') {
                renderMathInElement(document.body, {
                    delimiters: [
                        {left: '$$', right: '$$', display: true},
                        {left: '$', right: '$', display: false},
                        {left: '\\(', right: '\\)', display: false},
                        {left: '\\[', right: '\\]', display: true}
                    ],
                    throwOnError : false
                });
            }
            initChart();
            updateSimulator(5.5);
        };
